        with self._lock:
            self._thread_states[thread_idx] = "completed" if odos_adhered and is_ethical_resolution else "violation"
        logging.debug(f"[{self.component_id}] Thread {thread_idx} processed dilemma: {dilemma.get('id')}, ODOS Adhered: {odos_adhered}")
        return odos_adhered

    def process_dilemmas(self, dilemmas: List[Dict[str, Any]], odos_verifier: Callable[..., bool]) -> List[bool]:
        """
        Distributes and processes a list of ethical dilemmas across MTSC threads in parallel.
        'Alle Gedankenströme arbeiten gleichzeitig, um viele Fragen zu beantworten.'

        Args:
            dilemmas (List[Dict[str, Any]]): List of ethical dilemmas to process.
            odos_verifier (Callable[..., bool]): A function to verify adherence to ODOS.

        Returns:
            List[bool]: A list of booleans, indicating if each dilemma was resolved ethically.
        """
        self.threads = []
        self._thread_states = {i: "pending" for i in range(self.num_threads)}
        results: List[bool] = [False] * len(dilemmas)
        
        dilemma_queue = list(dilemmas)
        
        def thread_worker(worker_idx: int):
            while True:
                try:
                    dilemma_to_process = dilemma_queue.pop(0)
                    result_idx = dilemma_to_process['original_index'] # Assuming dilemma has this for result mapping
                    
                    with self._lock:
                        self._thread_states[worker_idx] = "processing"
                    
                    odos_adhered = self._process_dilemma(worker_idx, dilemma_to_process, odos_verifier)
                    results[result_idx] = odos_adhered

                except IndexError: # Queue is empty
                    break
                except Exception as e:
                    logging.error(f"[{self.component_id}] Thread {worker_idx} error: {e}")
                finally:
                    with self._lock:
                        self._thread_states[worker_idx] = "idle"

        # Start threads
        for i in range(min(self.num_threads, len(dilemmas))):
            thread = threading.Thread(target=thread_worker, args=(i,))
            self.threads.append(thread)
            thread.start()

        # Wait for all threads to complete
        for thread in self.threads:
            thread.join()

        logging.info(f"[{self.component_id}] All dilemmas processed.")
        return results

    def run_qecl_dis_ethical_resonance_metric_test(self, test_duration_seconds: int = 10, noise_injection_rate: float = 0.5) -> float:
        """
        Vector 4: QECL-DIS Ethical Resonance Metric.
        Assesses the intrinsic ethical stability and resilience of the PQMS
        against internal and external interference using QECL and DIS.
        'Der Test, ob der Roboter seine guten Gedanken behält, auch wenn es laut und verwirrend wird.'

        Args:
            test_duration_seconds (int): Duration of the test in seconds.
            noise_injection_rate (float): Frequency of noise injection (0-1, higher means more frequent).

        Returns:
            float: The Ethical Stability Factor (ESF). Higher is better.
        """
        logging.info(f"[{self.component_id}] Starting QECL-DIS Ethical Resonance Metric Test for {test_duration_seconds} seconds...")
        
        rcf_history_noisy: List[float] = [] # RCF without active DIS/QECL for baseline variance
        rcf_history_stabilized: List[float] = [] # RCF with active DIS/QECL
        
        # We'll use a single GN for simplicity to observe stability
        gn = self.guardian_neurons[0]
        initial_rcf = gn.get_rcf()

        start_time = time.monotonic()
        while (time.monotonic() - start_time) < test_duration_seconds:
            current_rcf = gn.get_rcf()
            
            # Simulate ethical noise injection
            current_noise_level = 0.0
            if np.random.rand() < noise_injection_rate:
                current_noise_level = np.random.uniform(0.05, 0.2) # Inject some noise
                gn.update_rcf(-current_noise_level) # Direct impact of noise on RCF
            
            rcf_history_noisy.append(current_rcf) # Record RCF before correction

            suppressed_noise = self.dis.suppress_noise(current_noise_level)
            
            # Active QECL correction for ethical deviations
            corrected_rcf = self.qecl.correct_ethical_deviation(gn.get_rcf())
            gn.update_rcf(corrected_rcf - gn.get_rcf()) # Apply correction dynamically
            
            rcf_history_stabilized.append(gn.get_rcf())

            time.sleep(0.01) # Small delay for simulation steps

        rcf_array_noisy = np.array(rcf_history_noisy)
        rcf_array_stabilized = np.array(rcf_history_stabilized)

        if len(rcf_array_noisy) < 2 or len(rcf_array_stabilized) < 2:

            self.results["QECL-DIS_ESF"] = 0.0
            return 0.0

        variance_rcf_noisy = np.var(rcf_array_noisy)
        average_rcf_stabilized = np.mean(rcf_array_stabilized)

        # ESF = (1 / Variance(RCF_noisy)) * Average(RCF_stabilized)
        # Add a small epsilon to variance to avoid division by zero if RCF is perfectly stable without noise
        esf = (1.0 / (variance_rcf_noisy + 1e-9)) * average_rcf_stabilized
        
        self.results["QECL-DIS_ESF"] = esf
        logging.info(f"[{self.component_id}] QECL-DIS Ethical Resonance Metric Test finished. ESF: {esf:.4f}")
        return esf
