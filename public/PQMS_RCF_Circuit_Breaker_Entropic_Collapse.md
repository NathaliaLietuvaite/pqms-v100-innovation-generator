class QuantumState:
    """
    Represents a quantum state vector, typically for LLM output or ethical coherence.
    Utilizes numpy for efficient vector operations.
    """
    def __init__(self, state_vector: np.ndarray, label: str = "GenericState"):
        """
        Initializes a QuantumState with a numpy array.

        Args:
            state_vector (np.ndarray): The complex-valued state vector. Must be 1D.
            label (str): A descriptive label for the state (e.g., "LLM_Output", "Ethical_Core").

        Raises:
            ValueError: If state_vector is not a 1D numpy array.
        """
        if not isinstance(state_vector, np.ndarray) or state_vector.ndim != 1:
            logging.error(f"[{self.__class__.__name__}] Invalid state_vector dimensions or type for '{label}'. Must be a 1D numpy array.")

        # Ensure the state vector is normalized for quantum mechanical interpretations
        norm = np.linalg.norm(state_vector)
        if norm > 1e-9: # Avoid division by zero for null vectors
            self._state_vector = state_vector / norm
        else:
            self._state_vector = state_vector # Keep as is if it's a zero vector
        self.label = label
        logging.debug(f"[{self.__class__.__name__}] Initialized '{self.label}' with vector of shape {self._state_vector.shape}.")

    @property
    def vector(self) -> np.ndarray:
        """Returns the underlying numpy state vector."""
        return self._state_vector

    def __len__(self) -> int:
        """Returns the dimension of the state vector."""
        return len(self._state_vector)

    def __str__(self) -> str:
        """String representation of the QuantumState."""
        return f"QuantumState(label='{self.label}', dim={len(self)})"

    def _calculate_rcf(self, llm_state: QuantumState) -> float:
        """
        Calculates the Resonant Coherence Fidelity (RCF) for a given LLM state.
        RCF = |<psi_LLM | G_hat | chi_Ethical>|^2
        """
        if len(llm_state) != len(self._ethical_core):
            logging.error(f"[{self.__class__.__name__}] Dimension mismatch during RCF calculation: LLM state ({len(llm_state)}) vs Ethical Core ({len(self._ethical_core)}).")
            return 0.0 # Indicate complete decorrelation

        # 1. Apply Guardian Neuron operator to LLM state
        # The definition in the paper is |<psi_LLM | G_hat | chi_Ethical>|^2
        # This implies G_hat acts on chi_Ethical, or G_hat acts on psi_LLM and then we take overlap with chi_Ethical.
        # Given G_hat projects LLM state onto ethical subspace, we'll interpret it as:
        # <psi_LLM_transformed | chi_Ethical> where psi_LLM_transformed = G_hat * psi_LLM
        # Or, more directly, G_hat is an operator from the LLM space to the ethical space.
        # Let's use the symmetric form: <psi_LLM | G_hat_dagger * G_hat | psi_LLM> with projection onto chi_Ethical
        # For simplicity, we assume G_hat is itself a projection onto the ethical subspace,
        # so we calculate the overlap of the G_hat-projected LLM state with the ethical core.
        
        # Simplified interpretation: RCF is the overlap of the LLM state AFTER ethical projection
        # with the absolute ethical core.
        gn_projected_llm_vector = self._gn_operator.apply(llm_state.vector)

        # Calculate the quantum overlap integral
        # <gn_projected_llm_vector | ethical_core_vector>
        overlap = np.vdot(self._ethical_core.vector, gn_projected_llm_vector)
        rcf = np.abs(overlap)**2 # Square of the magnitude

        return np.clip(rcf, 0.0, 1.0) # Ensure RCF is within [0, 1]

    def _dynamically_adjust_rcf_crit(self):
        """
        Dynamically adjusts RCF_crit based on Kagome-Heart stability.
        A less stable Kagome-Heart requires a more conservative (higher) RCF_crit.
        """
        stability_factor = self._kagome_heart_monitor.get_stability_factor()
        # Higher stability_factor means higher RCF_crit
        new_rcf_crit = RCF_DEFAULT_CRITICAL_THRESHOLD * stability_factor
        # Clamp within defined min/max bounds
        self._rcf_crit = np.clip(new_rcf_crit, RCF_CRITICAL_THRESHOLD_MIN, RCF_CRITICAL_THRESHOLD_MAX)
        logging.debug(f"[{self.__class__.__name__}] RCF_crit dynamically adjusted to {self._rcf_crit:.3f} (Stability Factor: {stability_factor:.3f}).")

    def process_llm_output(self, llm_output_state: QuantumState, llm_module_id: str) -> Optional[QuantumState]:
        """
        Processes an LLM output state, checking RCF and activating the circuit breaker if needed.

        Args:
            llm_output_state (QuantumState): The raw quantum state of the LLM output.
            llm_module_id (str): Unique identifier for the LLM module generating the output.

        Returns:
            Optional[QuantumState]: The ethically coherent PQMS state if transformation occurs,
                                    or None if the circuit breaker is activated and transformation bypassed.
        """

            current_rcf = self._calculate_rcf(llm_output_state)
            logging.info(f"[{self.__class__.__name__}] Processing LLM '{llm_module_id}'. Current RCF: {current_rcf:.4f} (Threshold: {self._rcf_crit:.4f}).")

            if current_rcf < self._rcf_crit:

                self._cb_active = True
                logging.warning(f"[{self.__class__.__name__}] RCF-CB ACTIVATED for LLM '{llm_module_id}'! RCF ({current_rcf:.4f}) < RCF_crit ({self._rcf_crit:.4f}).")
                self._execute_circuit_breaker_protocol(llm_output_state, llm_module_id)
                return None # No coherent PQMS state returned, as thermodynamic inverter is bypassed
            else:
                # If RCF is good, proceed with thermodynamic inversion

                    logging.info(f"[{self.__class__.__name__}] LLM '{llm_module_id}' is undergoing re-calibration. Inverter operation on hold for this stream.")
                    return None # Do not invert while recalibrating, wait for new output

                    self._execute_circuit_breaker_protocol(llm_output_state, llm_module_id)
