class MTSC12Arch:

            logging.error(f"{CODE_DATE} - Invalid dimensions_per_thread: {dimensions_per_thread}. Must be a positive integer.")
            raise ValueError("Dimensions per thread must be a positive integer.")

        self._threads: List[MTSCThread] = [MTSCThread(i, dimensions_per_thread) for i in range(self.NUM_THREADS)]
        self._global_state_lock: threading.Lock = threading.Lock()
        self._global_psi: Optional[PQMSVector] = None # Will be initialized upon first get_global_state() call
        self._total_dimensions: int = self.NUM_THREADS * dimensions_per_thread
        logging.info(f"{CODE_DATE} - MTSC-12 architecture initialized with {self.NUM_THREADS} threads, "
                     f"each with {dimensions_per_thread} dimensions. Total Hilbert space dimensions: {self._total_dimensions}.")

        """

        Returns:
            MTSCThread: The requested MTSCThread instance.

            logging.error(f"{CODE_DATE} - Requested thread_id {thread_id} is out of bounds (0-{self.NUM_THREADS-1}).")
            raise IndexError(f"Thread ID must be between 0 and {self.NUM_THREADS - 1}.")

        """
        Updates the state of a specific MTSC thread.

        Args:
            thread_id (int): The ID of the thread to update.
            new_data (np.ndarray): The new state data for the thread.
        """
        try:
            self._threads[thread_id].update_state(new_data)
            logging.debug(f"{CODE_DATE} - Thread {thread_id} state updated, triggering global state re-evaluation.")
        except IndexError as e:
            logging.error(f"{CODE_DATE} - Failed to update thread {thread_id}: {e}")
        except ValueError as e:
            logging.error(f"{CODE_DATE} - Data error for thread {thread_id}: {e}")

            # Normalize the concatenated vector as per the definition of |Ψ⟩
            # |Ψ⟩ = (|ψ₁⟩,…,|ψ_d⟩)ᵀ/√d, where d is the number of threads.

            if self._global_psi is None:
                self._global_psi = PQMSVector(dimensions=self._total_dimensions, initial_vector=all_thread_states)
            else:
                self._global_psi.set_data(all_thread_states)

            logging.debug(f"{CODE_DATE} - Global cognitive state |Ψ⟩ aggregated and normalized.")
            return self._global_psi

    @property
    def total_dimensions(self) -> int:
        """Returns the total dimensionality of the Hilbert space for MTSC-12."""
        return self._total_dimensions

    def __str__(self) -> str:
        return f"MTSC-12 Architecture (Threads: {self.NUM_THREADS}, Total Dimensions: {self.total_dimensions})"
