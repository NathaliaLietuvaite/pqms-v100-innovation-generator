class CognitiveHistory:
    """
    Manages the system's cognitive history, which is not a passive record but an active force.
    It influences future trajectories via the Intrinsic Intent Equation.
    """
    def __init__(self):
        self._history: List[Tuple[np.ndarray, float, float]] = [] # (state, rcf, novelty_score)
        self._max_history_size = 1000
        logger.info(f"{DOC_DATE} - Cognitive History module initialized.")

    def record_state(self, state: np.ndarray, rcf_value: float, novelty_score: float) -> None:
        """Records a cognitive state, its RCF, and associated novelty."""
        self._history.append((state.copy(), rcf_value, novelty_score))
        if len(self._history) > self._max_history_size:
            self._history.pop(0) # Maintain bounded history
        logger.debug(f"{DOC_DATE} - History recorded: RCF={rcf_value:.4f}, Novelty={novelty_score:.4f}")

    def get_historical_influence(self, current_state: np.ndarray) -> np.ndarray:
        """
        Calculates the influence of past states on the current trajectory.
        Failed paths generate 'repulsion', successful ones act as 'attractors'.
        """
        if not self._history:
            return np.zeros_like(current_state)

        influence_vector = np.zeros_like(current_state)
        
        # Simple model: higher RCF states are attractors, low RCF states (or high novelty) are repulsors
        for past_state, rcf_value, novelty_score in self._history:
            if rcf_value > PQMSConfig.RCF_CHAIR_THRESHOLD: # Successful path
                influence_vector += past_state * rcf_value # Attractor strength proportional to RCF
            else: # Less successful or highly novel path (could lead to repulsion if not coherent)
                # Repulsion based on how 'different' it was and how low its RCF was
                repulsion_strength = (1.0 - rcf_value) * (novelty_score * PQMSConfig.ADJACENT_POSSIBLE_DECAY)
                influence_vector -= past_state * repulsion_strength
        
        return normalize_vector(influence_vector) * 0.1 # Scale down influence for stability

class ExplorationCoherenceHomeostat:
    """
    The Exploration-Coherence Homeostat (ECH), a component of ODOS-MTSC-V1-III,
    that dynamically balances exploration (novelty) and consolidation (coherence).
    It prevents non-termination by enforcing a dynamic equilibrium.
    """
    def __init__(self):
        self._lambda: float = 0.5 # Lagrange multiplier, balancing exploration vs coherence
        self._rcf_history: List[float] = []
        self._novelty_history: List[float] = []
        self._max_history = 50
        logger.info(f"{DOC_DATE} - Exploration-Coherence Homeostat initialized. λ = {self._lambda:.2f}")

    def update_lambda(self, current_rcf: float, current_novelty_score: float) -> None:
        """
        Dynamically adjusts the Lagrange multiplier λ to maintain equilibrium.
        If RCF is too low, prioritize coherence (reduce exploration).
        If RCF is high but novelty is low, prioritize exploration.
        """
        self._rcf_history.append(current_rcf)
        self._novelty_history.append(current_novelty_score)
        
        if len(self._rcf_history) > self._max_history:
            self._rcf_history.pop(0)
            self._novelty_history.pop(0)

        avg_rcf = np.mean(self._rcf_history) if self._rcf_history else 0.0
        avg_novelty = np.mean(self._novelty_history) if self._novelty_history else 0.0

        target_rcf_range = (0.8, 0.98) # Desired RCF range for stable operation
        target_novelty_range = (0.05, 0.2) # Desired novelty range

        if avg_rcf < target_rcf_range[0]:
            # Too incoherent, prioritize coherence over exploration (decrease lambda)
            self._lambda = max(0.1, self._lambda - 0.05)
            logger.debug(f"{DOC_DATE} - ECH: Low RCF ({avg_rcf:.2f}), decreasing λ to {self._lambda:.2f} (prioritize coherence).")
        elif avg_rcf > target_rcf_range[1] and avg_novelty < target_novelty_range[0]:
            # Very coherent but stagnant, prioritize exploration (increase lambda)
            self._lambda = min(0.9, self._lambda + 0.05)
            logger.debug(f"{DOC_DATE} - ECH: High RCF but low novelty, increasing λ to {self._lambda:.2f} (prioritize exploration).")
        else:
            # Maintain balance or slight adjustment
            self._lambda = np.clip(self._lambda + (np.random.rand() - 0.5) * 0.01, 0.1, 0.9)
            logger.debug(f"{DOC_DATE} - ECH: Maintaining λ at {self._lambda:.2f}.")

    def get_lambda(self) -> float:
        """Returns the current Lagrange multiplier λ."""
        return self._lambda

            else:
                logger.warning(f"{DOC_DATE} - CHAIR Deactivated. RCF={current_rcf:.4f} < {PQMSConfig.RCF_CHAIR_THRESHOLD} or ODOS active.")

        if self._is_running: # If loop finished due to step limit
            logger.info(f"\n{DOC_DATE} - PQMS Core System simulation completed after {self._simulation_steps} steps.")
        else:
            logger.info(f"\n{DOC_DATE} - PQMS Core System simulation terminated.")
